import React, { useCallback, useRef } from "react";
import { AgGridReact } from "ag-grid-react";
import "ag-grid-community/styles/ag-grid.css";
import "ag-grid-community/styles/ag-theme-alpine.css";
import {
  Col,
  OverlayTrigger,
  Row,
  Tooltip,
} from "react-bootstrap";
import ExportIcon from "../../Assets/Images/export.png";

import "./aggrid.css";

/**
 * DataTable
 * @param {any} columnDefs column Definition
 * @param {any} rowData row Data
 * @returns {any} AgGrid Table
 */
const DataTable = ({ columnDefs, rowData }) => {
  /* State Declarations*/
  const gridRef = useRef();

  /* Function Declarations */

  /* This function display tool tip on export button */
  const renderExport = (props) => (
    <Tooltip style={{ background: "red" }} id="button-tooltip" {...props}>
      Export Data
    </Tooltip>
  );

  /* Display confirmation message on dataTable export */
  const onBtnExport = useCallback(() => {
    const confirm = window.confirm(
      `Are you sure to download data for TableName?`,
    );
    if (confirm) {
      gridRef.current.api.exportDataAsCsv({ fileName: "Downloaded" });
    }
  }, []);


  return (
    <div className="container-fluid py-3">
      {rowData.length > 0 ? (
        <div
          className="d-flex justify-content-end mb-2"
          style={{ gap: "1rem" }}
        >
          <OverlayTrigger placement="top" overlay={renderExport}>
            <div className="export-button" onClick={onBtnExport}>
              <Row>
                <Col md="2">
                  <div className="icon">
                    <img src={ExportIcon} alt="Export" />
                  </div>
                </Col>
              </Row>
            </div>
          </OverlayTrigger>
        </div>
      ) : (
        ""
      )}
      <div className="ag-theme-alpine" style={{ height: 400 }}>
        <AgGridReact
          ref={gridRef}
          rowData={rowData}
          columnDefs={columnDefs}
        ></AgGridReact>
      </div>
    </div>
  );
};

export default React.memo(DataTable);
