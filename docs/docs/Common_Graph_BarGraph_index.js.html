import React from "react";
import Plot from "react-plotly.js";
/**
 * 
 * @param {any} graphData 
 * @param {any} specificLayoutProperties 
 * @param {any} handleGraphClick
 * @returns Plot
 */
const BarGraph = ({
  graphData,
  specificLayoutProperties,
  handleGraphClick,
}) => {
  /* Graph Layout */
  const layout = {
    title:
      specificLayoutProperties && specificLayoutProperties.graphTitle ?
        specificLayoutProperties.graphTitle :
        "",
    barmode: "group",
    bargap:
      specificLayoutProperties && specificLayoutProperties.bargap ?
        specificLayoutProperties.bargap :
        "",
    width: specificLayoutProperties && specificLayoutProperties.width ?
      specificLayoutProperties.width :
      "",
    height: 450,
    margin:
      specificLayoutProperties && specificLayoutProperties.margin ?
        specificLayoutProperties.margin :
        {},
    paper_bgcolor: "FFFFFF", // '#F8F8F8',
    plot_bgcolor: "FFFFFF", // '#F8F8F8',
    showlegend:
      specificLayoutProperties && specificLayoutProperties.showlegend ?
        "false" :
        "true",

    xaxis: {
      title:
        specificLayoutProperties && specificLayoutProperties.x_axis_title ?
          specificLayoutProperties.x_axis_title :
          "",
      automargin: true,
      tickangle: 270,
      linecolor: "transparent",
      linewidth: 10,
    },
    yaxis: {
      title:
        specificLayoutProperties && specificLayoutProperties.y_axis_title ?
          specificLayoutProperties.y_axis_title :
          "",
    },
    hovermode: "closest",
    hoverlabel: { bgcolor: "#FFF" },
    dragmode: true,
    legend: {
      itemclick: "toggleothers",
      traceorder: "reversed",
    },
    shapes:
      specificLayoutProperties && specificLayoutProperties.shapes ?
        specificLayoutProperties.shapes :
        [],
  };

  return (
    <div>
      <Plot
        data={graphData}
        layout={layout}
        onClick={handleGraphClick}
      />
    </div>
  );
};

export default React.memo(BarGraph);
