import moment from "moment";

/**
 * Description
 * @param {any} requiredDate
 * @returns {any} dateFullFormatCheck
 */
const DateFormatter = (requiredDate) => {
  let dateFullFormatCheck = "";
  const hntsUser = JSON.parse(localStorage.getItem("hntsUser"));
  const requiredDateFormat = hntsUser && hntsUser.settings && hntsUser.settings.date_format ? hntsUser.settings.date_format : "YYYY/MM/DD";
  // If Date length is greater thatis if it is with time we will use moment;
  if (requiredDate && requiredDate.length > 0) {
    dateFullFormatCheck = moment(requiredDate).format(requiredDateFormat);
  } else {
    dateFullFormatCheck = requiredDate;
  }
  return (
    dateFullFormatCheck
  );
};

export default DateFormatter;
