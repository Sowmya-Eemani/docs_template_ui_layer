import React from "react";
import { Dropdown } from "react-bootstrap";
import { Link } from "react-router-dom";

/**
 * Description
 * @param {any} {dropDownHeader
 * @param {any} dropDownList
 * @param {any} type
 * @param {any} handleClick}
 * @returns {any} DropDown
 */
const DropDown = ({ dropDownHeader, dropDownList, type, handleClick }) => {
  let dropDownValue = "";
  return (
    <div>
      <Dropdown>
        <Dropdown.Toggle variant="success" id="dropdown-basic">
          {dropDownHeader ? dropDownHeader : ""}
        </Dropdown.Toggle>

        {type ? (
          <Dropdown.Menu>
            {dropDownList ? (
              dropDownList.map((value, index) => {
                return (
                  <Dropdown.Item onClick={() => {
                    handleClick(value);
                  }}>
                    {value}
                  </Dropdown.Item>
                );
              })
            ) : (
              <Dropdown.Item>No Data</Dropdown.Item>
            )}
          </Dropdown.Menu>
        ) : (
          <Dropdown.Menu>
            {dropDownList ? (
              dropDownList.map((value, index) => {
                dropDownValue = "/" + value;
                return (
                  <Dropdown.Item as={Link} to={dropDownValue}>
                    {value.toUpperCase()}
                  </Dropdown.Item>
                );
              })
            ) : (
              <Dropdown.Item>No Data</Dropdown.Item>
            )}
          </Dropdown.Menu>
        )}
      </Dropdown>
    </div>
  );
};

export default React.memo(DropDown);
